🚀 Production Deploy Guide

Architecture for Real Multiplayer

This demo works with AI + BroadcastChannel (same-device tabs). For true cross-device multiplayer, deploy this stack:

Tech Stack

  • Frontend: This HTML (as-is) + Socket.IO client
  • Backend: Node.js + Express + Socket.IO
  • Database: Supabase (PostgreSQL) or Redis
  • Hosting: Railway.app or Render.com

Server Code (server.js)

const express = require('express');
const http = require('http');
const { Server } = require('socket.io');
const app = express();
const server = http.createServer(app);
const io = new Server(server, { cors: { origin: '*' } });

let waitingPlayer = null;
const rooms = new Map();

io.on('connection', (socket) => {
  socket.on('find_match', (name) => {
    if (waitingPlayer && waitingPlayer.id !== socket.id) {
      const roomId = 'room_' + Date.now();
      const room = {
        id: roomId,
        playerX: { id: waitingPlayer.id, name: waitingPlayer.name },
        playerO: { id: socket.id, name },
        board: Array(9).fill(null),
        turn: 'X', status: 'playing'
      };
      rooms.set(roomId, room);
      waitingPlayer.socket.join(roomId);
      socket.join(roomId);
      io.to(roomId).emit('match_found', room);
      waitingPlayer = null;
    } else {
      waitingPlayer = { id: socket.id, name, socket };
    }
  });

  socket.on('answer_question', ({ roomId, cell, correct }) => {
    const room = rooms.get(roomId);
    if (!room) return;
    // Server validates turn & cell
    if (correct) {
      room.board[cell] = room.turn;
      io.to(roomId).emit('move_made', { cell, symbol: room.turn });
      // Check win/draw then switch turn
    }
    room.turn = room.turn === 'X' ? 'O' : 'X';
    io.to(roomId).emit('turn_change', room.turn);
  });
});

server.listen(3000);

Database Schema (Supabase SQL)

CREATE TABLE players (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  wins INT DEFAULT 0,
  losses INT DEFAULT 0,
  draws INT DEFAULT 0,
  online BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE questions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  question TEXT NOT NULL,
  choice_a TEXT, choice_b TEXT,
  choice_c TEXT, choice_d TEXT,
  correct_answer CHAR(1) NOT NULL
);

CREATE TABLE matches (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  player_x UUID REFERENCES players(id),
  player_o UUID REFERENCES players(id),
  winner UUID REFERENCES players(id),
  result TEXT,
  played_at TIMESTAMPTZ DEFAULT now()
);
